#ifdef //If already set up then it will skip set up again. 
#define F_CPU 16000000L //Sets the clock speed to 16Mhz utilizing the on board crystal on the arduino. 
#endif

#include <avr/io.h> //Brings in the IO library necessary for the Port numbers on the arduino. You may view this file under the dependencies. 
#include <util/delay.h> //Brings in the library with the delay function. 


int main(void) 
{
	//Anthing outside of the while(1) loop is the equivalent to void setup() THIS IS ONLY RUN ONCE
	DDRB |= (1<<PB5); //Sets Pin 5 on Port B as output -> 00100000
    
    while (1) //Begins infinite loop this is the equivelent to Arduino IDE Void Loop()
    {
		PORTB |= (1<<PB5); //PORTB is a resistor with 8 bits XXXXXXXX the | (OR) Operator will set the 5th bit 00100000 then writing to register XX1XXXXX
		_delay_ms(500); //Delay function that waits in Milliseconds
		PORTB &= ~(1<<PB5); //PORTB is a resistor with 8 bits XXXXXXXX the & (AND) operator will set the 5th bit opposite 11011111 then writing to register XX0XXXXX 
		_delay_ms(500);
		//Try reducing this code by 2 lines!
    }
}
